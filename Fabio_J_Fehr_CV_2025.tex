% Minimalist CV Template
% Clean typography optimized for human readers and ATS systems
% No photos, highly customizable sections with toggle functionality

\documentclass[11pt,a4paper,sans]{moderncv}

% Modern CV style and color scheme
\moderncvstyle{classic}
\moderncvcolor{black}

% Character encoding
\usepackage[utf8]{inputenc}

% Page margins - optimized for readability
\usepackage[scale=0.85]{geometry}
\setlength{\hintscolumnwidth}{3cm}

% Section spacing - more air between sections
\usepackage{etoolbox}
\patchcmd{\section}{1ex}{2ex}{}{}

% Make section headers stand out more
\renewcommand*{\sectionfont}{\Large\bfseries\upshape}

% Custom colours - must come after moderncv style
\definecolor{textgrey}{RGB}{20,20,20}
\AtBeginDocument{\color{textgrey}}

% Custom command for better spacing control
\newcommand{\cvitemcompact}[2]{%
  \cvitem{#1}{#2}%
}

% Smaller bullet points for cleaner minimalist look
\renewcommand{\labelitemi}{\small$\bullet$}

% Load user content
\input{cv-content}

% ===============================================
% DOCUMENT START
% ===============================================

\begin{document}

\makecvtitle

% ===============================================
